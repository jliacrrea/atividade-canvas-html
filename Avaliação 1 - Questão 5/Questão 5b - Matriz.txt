      for(let i=0; i<5; i++){
        for(let j=0; j<5; j++){

          // Quadrado
          ctx.strokeRect(20 + (120 * i), 20 + (120 * j), 100, 100);
          
          // TriÃ¢ngulo
          ctx.beginPath();
          ctx.moveTo(110 + (120 * i), 95 + (120 * j));
          ctx.lineTo(30 + (120 * i), 95 + (120 * j));
          ctx.lineTo(70 + (120 * i), 40 + (120 * j));
          ctx.closePath();
          ctx.stroke();
        }
      }
